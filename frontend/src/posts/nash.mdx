import random

moves = ["rock", "paper", "scissors"]
payoffs = {("rock", "rock"): (0, 0), ("rock", "scissors"): (1, -1), ... }

def simulate(n=10000):
    p1 = p2 = 0
    for _ in range(n):
        m1, m2 = random.choice(moves), random.choice(moves)
        s1, s2 = payoffs[(m1, m2)]
        p1 += s1; p2 += s2
    return p1, p2

simulate()
